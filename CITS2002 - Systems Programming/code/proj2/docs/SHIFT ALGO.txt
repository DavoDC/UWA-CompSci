#include <stdio.h>

int
main ()
{
  printf ("\n Hello World");

  // Data
  int len = 2;
  int entries[] = { 1, 2 };
  int remPos = 0;

  // print all
  printf ("\n");
  for (int i = 0; i < len; i++)
    {
      printf (" val: %i", entries[i]);
    }

  printf ("\n");

  // remove
  for (int n = remPos; n < len; n++)
    {
      //printf ("\n %i <- %i", entries[n], entries[n + 1]);
      entries[n] = entries[n + 1];
    }

  // print all
  printf ("\n");
  for (int i = 0; i < len; i++)
    {
      printf (" val: %i", entries[i]);
    }

  return 0;
}
