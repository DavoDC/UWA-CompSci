# one process, no I/O, process executes for several time quanta before it exits.
# Run with 	/cslinux/bin/besttq-sample -v tfB 50
#
process 1 10 {
    exit        300
}

# Using the web-interface
# T: 0 = Reboot with TQ = 50
# T: 10 = Process 1 moves from New to Ready and CS starts
# T: 15 = Process 1 moves from Ready to Running
# T: 65 = Process 1 remains in Running queue
# T: 115 = Process 1 remains in the Running queue
# T: 165 = Process 1 remains in the Running queue
# T: 215 = Process 1 remains in the Running queue
# T: 265 = Process 1 remains in the Running queue
# T: 315 = Process 1 ends and moves from Running to Exit
# best 50 305
# note the total time that process 1 is in the Running queue is 300us (which is the exit time)
